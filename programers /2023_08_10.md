def solution(numbers):
    
    answer = []
    
    while numbers:
        
        for i in range(len(numbers)):
            if max(numbers) == numbers[i]:
                answer.append(-1)
            else:
                for j in range(i, len(numbers)):
                    if numbers[j] > numbers[i]:
                        answer.append(numbers[j])
                        break
            break
        numbers.pop(0)
    
    return answer
